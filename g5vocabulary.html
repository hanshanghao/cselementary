        function shuffle(array) {

            const arr = [...array];

            for (let i = arr.length - 1; i > 0; i--) {

                const j = Math.floor(Math.random() * (i + 1));

                [arr[i], arr[j]] = [arr[j], arr[i]];

            }

            return arr;

        }

                        const displayContent = isEmoji 

                            ? `<div class="card-emoji">${cardData.content}</div>`

                            : `<div class="card-text">${cardData.content}</div>`;

                        

                        card.innerHTML = `

                            <div class="card-inner">

                                <div class="card-face card-front">?</div>

                                <div class="card-face card-back">${displayContent}</div>

                            </div>

                        `;

                    }

                }


                card.onclick = () => flipCard(card, i);

                board.appendChild(card);

            }

        }


        async function startGame() {

            if (!currentUnit) {

                alert('Please select a lesson first! 請先選擇單元！');

                return;

            }


            moves = 0;

            matchedPairs = 0;

            flippedCards = [];

            document.getElementById('moves').textContent = '0';

            document.getElementById('matches').textContent = '0';


            generateGameBoard();

            displayBoard();


            await showCountdown();

            await peekCards();

            

            gameStarted = true;

        }


        async function showCountdown() {

            const overlay = document.getElementById('countdownOverlay');

            const number = document.getElementById('countdownNumber');

            

            for (let i = 3; i > 0; i--) {

                number.textContent = i;

                overlay.classList.add('show');

                await sleep(1000);

                overlay.classList.remove('show');

                await sleep(100);

            }

        }


        async function peekCards() {

            const cards = document.querySelectorAll('.flip-card');

            cards.forEach(card => card.classList.add('flipped'));

            await sleep(3000);

            cards.forEach(card => card.classList.remove('flipped'));

        }


        function flipCard(card, index) {

            if (!gameStarted || !canFlip) return;

            if (card.classList.contains('flipped') || card.classList.contains('matched')) return;

            if (flippedCards.length >= 2) return;


            if (index === minePosition) {

                card.classList.add('flipped');

                card.classList.add('mine-hit');

                hitMine();

                return;

            }


            card.classList.add('flipped');

            flippedCards.push(card);


            if (flippedCards.length === 2) {

                moves++;

                document.getElementById('moves').textContent = moves;

                setTimeout(checkMatch, 800);

            }

        }


        function checkMatch() {

            const [card1, card2] = flippedCards;

            const pair1 = card1.dataset.pairId;

            const pair2 = card2.dataset.pairId;


            if (pair1 === pair2) {

                card1.classList.add('matched');

                card2.classList.add('matched');

                matchedPairs++;

                document.getElementById('matches').textContent = matchedPairs;


                if (matchedPairs === 4) {

                    setTimeout(showSuccess, 500);

                }

            } else {

                setTimeout(() => {

                    card1.classList.remove('flipped');

                    card2.classList.remove('flipped');

                }, 1000);

            }


            flippedCards = [];

        }


        async function hitMine() {

            canFlip = false;

            const overlay = document.getElementById('freezeOverlay');

            const countdown = document.getElementById('freezeCountdown');

            

            overlay.classList.add('show');


            if (navigator.vibrate) {

                navigator.vibrate([200, 100, 200]);

            }


            for (let i = 3; i > 0; i--) {

                countdown.textContent = i;

                await sleep(1000);

            }


            overlay.classList.remove('show');

            

            const cards = document.querySelectorAll('.flip-card');

            cards[minePosition].classList.remove('flipped');

            cards[minePosition].classList.remove('mine-hit');

            

            canFlip = true;

        }


        function showSuccess() {

            const stats = `Moves: ${moves} | 移動次數: ${moves}`;

            document.getElementById('successStats').textContent = stats;

            document.getElementById('successOverlay').classList.add('show');

        }


        function closeSuccess() {

            document.getElementById('successOverlay').classList.remove('show');

        }


        function restartRound() {

            if (!gameStarted) return;

            

            moves = 0;

            matchedPairs = 0;

            flippedCards = [];

            document.getElementById('moves').textContent = '0';

            document.getElementById('matches').textContent = '0';


            const cards = document.querySelectorAll('.flip-card');

            cards.forEach(card => {

                card.classList.remove('flipped', 'matched', 'mine-hit');

            });


            peekCards();

        }


        function newRound() {

            if (!currentUnit) {

                alert('Please select a lesson first! 請先選擇單元！');

                return;

            }

            startGame();

        }


        function sleep(ms) {

            return new Promise(resolve => setTimeout(resolve, ms));

        }
